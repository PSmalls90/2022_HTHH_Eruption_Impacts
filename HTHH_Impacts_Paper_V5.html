theme_nature <- function(base_size = 6) {
  theme_classic(base_size = base_size) +
    theme(
      text = element_text(family = base_family),
      axis.title = element_text(size = base_size),
      axis.text  = element_text(size = base_size),
      axis.line  = element_line(linewidth = 0.4),
      axis.ticks = element_line(linewidth = 0.4),
      legend.title = element_text(size = base_size),
      legend.text  = element_text(size = base_size),
      legend.key.size = unit(3, "mm"),
      plot.title = element_text(size = base_size + 1, face = "bold"),
      plot.margin = margin(4, 4, 4, 4, unit = "mm")
    )
}

fit_or_load_brm <- function(path, formula, data, family, prior,
                            chains = 4, iter = 4000,
                            control = list(adapt_delta = 0.95, max_treedepth = 15),
                            seed = 123) {
  if (file.exists(path)) {
    readRDS(path)
  } else {
    fit <- brm(
      formula = formula,
      data = data,
      family = family,
      prior = prior,
      chains = chains,
      iter = iter,
      control = control,
      seed = seed
    )
    saveRDS(fit, path)
    fit
  }
}

summarise_epred <- function(pred, newdata) {
  stopifnot(ncol(pred) == nrow(newdata))
  newdata %>%
    mutate(
      pred_median = apply(pred, 2, median),
      lower = apply(pred, 2, quantile, probs = 0.025),
      upper = apply(pred, 2, quantile, probs = 0.975)
    )
}

unscale <- function(x_sc, x_raw) x_sc * sd(x_raw, na.rm = TRUE) + mean(x_raw, na.rm = TRUE)

# (A.1) Time × management Raw Data Plot
.time_mgmt_raw <- function(df) {
  df_plot <- df %>%
    filter(management_secondary_name %in% c("No-Take", "Open")) %>%
    mutate(
      pre_post = factor(pre_post, levels = c("pre", "post"),
                        labels = c("Pre-Eruption", "Post-Eruption")),
      management_secondary_name = factor(management_secondary_name, levels = c("No-Take", "Open")),
      biomass_bt = expm1(target_biomass_log1p)
    )

  sum_df <- df_plot %>%
    group_by(management_secondary_name, pre_post) %>%
    summarise(
      mean_bt = mean(biomass_bt, na.rm = TRUE),
      se_bt   = sd(biomass_bt, na.rm = TRUE) / sqrt(sum(!is.na(biomass_bt))),
      lower   = mean_bt - 1.96 * se_bt,
      upper   = mean_bt + 1.96 * se_bt,
      .groups = "drop"
    ) %>%
    mutate(lower = pmax(lower, 0))

  ggplot(sum_df,
         aes(x = pre_post, y = mean_bt,
             group = management_secondary_name,
             color = management_secondary_name)) +
    geom_errorbar(aes(ymin = lower, ymax = upper), width = 0.10, linewidth = 0.6) +
    geom_point(size = 1.6) +
    geom_line(linewidth = 0.6) +
    scale_color_manual(values = c("No-Take" = "#2E8B57", "Open" = "#1F5AA6"), name = "") +
    labs(x = "Time period", y = "Target species biomass (kg/ha)") +
    scale_y_continuous(trans = "sqrt") +
    theme_nature(base_size = 5) +
    theme(legend.position = c(0.80, 0.80))
}

# (A.2) Time × management Predcited Data Plot
.time_mgmt_pred <- function(model, df) {
  nd <- crossing(
    pre_post = factor(c("pre", "post"), levels = c("pre", "post")),
    management_secondary_name = factor(c("No-Take", "Open"), levels = c("No-Take", "Open"))
  ) %>%
    mutate(
      distance_sc = mean(df$distance_sc, na.rm = TRUE),
      depth_sc    = mean(df$depth_sc, na.rm = TRUE),
      fp_sc       = mean(df$fp_sc, na.rm = TRUE)
    )

  p <- posterior_epred(model, newdata = nd, re_formula = NA)
  p_bt <- expm1(p)

  sum_df <- nd %>%
    mutate(
      mean_bt = apply(p_bt, 2, median),
      lower   = apply(p_bt, 2, quantile, 0.025),
      upper   = apply(p_bt, 2, quantile, 0.975),
      pre_post = factor(pre_post, levels = c("pre", "post"),
                        labels = c("Pre-Eruption", "Post-Eruption"))
    ) %>%
    mutate(lower = pmax(lower, 0))

  ggplot(sum_df,
         aes(x = pre_post, y = mean_bt,
             group = management_secondary_name,
             color = management_secondary_name)) +
    geom_errorbar(aes(ymin = lower, ymax = upper), width = 0.10, linewidth = 0.6) +
    geom_point(size = 1.6) +
    geom_line(linewidth = 0.6) +
    scale_color_manual(values = c("No-Take" = "#2E8B57", "Open" = "#1F5AA6"), name = "") +
    labs(x = "Time period", y = "Target species biomass (kg/ha)") +
    scale_y_continuous(trans = "sqrt") +
    theme_nature(base_size = 5) +
    theme(legend.position = c(0.80, 0.80))
}
